JavaScript for Progress Bar (in script.js):

javascript
// ADD scroll progress tracking
document.addEventListener('DOMContentLoaded', function() {
    const carousel = document.querySelector('.carousel-container-peek');
    const progress = document.querySelector('.progress-indicator');
    
    if (carousel && progress) {
        carousel.addEventListener('scroll', function() {
            const maxScroll = carousel.scrollWidth - carousel.clientWidth;
            const currentScroll = carousel.scrollLeft;
            const progressPercent = (currentScroll / maxScroll) * 100;
            progress.style.width = Math.max(20, progressPercent) + '%';
        });
    }
});
